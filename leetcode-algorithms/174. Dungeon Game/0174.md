> 第一种思路

```python
class Solution:
    def kthSmallest(self, root: TreeNode, k: int) -> int:
        if not root: return
        #中序遍历,二叉搜索树BST的中序遍历是有序的，从小到大
        stack,node,count = [],root,0
        while stack or node:
            while node:
                stack.append(node)
                node = node.left
            node = stack.pop()
            count += 1
            if count == k: return node.val
            node = node.right
```
